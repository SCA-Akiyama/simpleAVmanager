import { deviceManager } from "./lib/manager";
import { initCronJobs } from "./lib/cron";
import { handleClientMessage } from "./lib/ws-router";
import { closeDb } from "./lib/db";

deviceManager.init();
initCronJobs();

const server = Bun.serve({
  port: 3000,
  async fetch(req, server) {
    const url = new URL(req.url);

    // 1. WebSocket縺ｸ縺ｮ繧｢繝・・繧ｰ繝ｬ繝ｼ繝峨Μ繧ｯ繧ｨ繧ｹ繝医ｒ蜃ｦ逅・
    if (url.pathname === "/ws") {
      const success = server.upgrade(req);
      if (success) return undefined; // Bun縺係S謗･邯壹∈蛻・ｊ譖ｿ縺医ｋ
    }

  },

  websocket: {
    open(ws) {
      ws.subscribe("updates");
      ws.send(JSON.stringify({ type: "INIT", data: deviceManager.getStatus() }));
    },
    async message(ws, msg) {
      // 蜈ｨ蜩｡縺ｸ縺ｮ荳譁蛾∽ｿ｡
      const broadcastUpdate = () => {
        server.publish("updates", JSON.stringify({ type: "SYNC", data: deviceManager.getStatus() }));
      };
      
      // 庁 縺薙・繧ｯ繝ｩ繧､繧｢繝ｳ繝医□縺代∈縺ｮ霑比ｿ｡
      const sendResponse = (data: any) => {
        ws.send(JSON.stringify(data));
      };
      
      await handleClientMessage(msg.toString(), broadcastUpdate, sendResponse);
    },
  }
});

deviceManager.startPolling(() => {
  server.publish("updates", JSON.stringify({ type: "SYNC", data: deviceManager.getStatus() }));
});

console.log(`噫 Server started at http://localhost:${server.port}`);

const shutdown = () => {
  console.log("\n尅 繧ｵ繝ｼ繝舌・縺ｮ繧ｷ繝｣繝・ヨ繝繧ｦ繝ｳ繧帝幕蟋九＠縺ｾ縺・..");

  closeDb();
  process.exit(0);
};

// src/lib/cron.ts
import { deviceManager } from "./manager";
import { scheduleDb , logDb} from "./db";

// 庁 螳溯｡御ｸｭ縺ｮCron繧ｸ繝ｧ繝悶ｒ菫晄戟縺吶ｋMap繧定ｿｽ蜉
export const activeCronJobs = new Map<string, any>();

export const scheduleRecurring = (cronExp: string, ids: string[], patch: any, id: string = crypto.randomUUID()) => {
  scheduleDb.add(id, "cron", cronExp, ids, patch);
  
  // 庁 謌ｻ繧雁､縺ｮ繧ｸ繝ｧ繝悶う繝ｳ繧ｹ繧ｿ繝ｳ繧ｹ繧貞､画焚縺ｧ蜿励￠蜿悶ｋ
  const job = Bun.cron(cronExp, () => {
    logDb.add("INFO", "CRON", "EXECUTE_CRON", { cronExp, patch }, ids.join(","));
    // 窶ｻ Map縺ｧ繧ｹ繝医ャ繝励＆縺帙ｋ縺溘ａ縲∵ｯ主屓縺ｮDB逕溷ｭ倡｢ｺ隱・exists)縺ｯ荳崎ｦ√↓縺ｪ繧翫∪縺呻ｼ・
    deviceManager.updateDesired(ids, patch);
  });
  
  // 庁 Map縺ｫ菫晏ｭ倥☆繧・
  activeCronJobs.set(id, job);
  console.log(`竢ｰ [螳壽悄逋ｻ骭ｲ] ${cronExp} -> ${ids.join(",")}`);
};

// 庁 蜑企勁逕ｨ縺ｮ髢｢謨ｰ繧定ｿｽ蜉
export const cancelCronJob = (id: string) => {
  const job = activeCronJobs.get(id);
  if (job) {
    job.stop(); // 繝｡繝｢繝ｪ縺九ｉ螳悟・縺ｫ蜑企勁
    activeCronJobs.delete(id);
    console.log(`尅 [螳壽悄隗｣髯､] 繝｡繝｢繝ｪ荳翫・繧ｸ繝ｧ繝悶ｒ蛛懈ｭ｢縺励∪縺励◆: ${id}`);
  }
};

/**
 * 庁 蜊倡匱繧ｸ繝ｧ繝悶・逋ｻ骭ｲ
 */
export const scheduleOnce = (date: Date, ids: string[], patch: any, id: string = crypto.randomUUID()) => {
  const timeMs = date.getTime();
  const now = Date.now();
  if (timeMs <= now) return console.warn("笞・・驕主悉縺ｮ譌･譎ゅ・繧ｹ繧ｱ繧ｸ繝･繝ｼ繝ｫ縺ｧ縺阪∪縺帙ｓ");

  scheduleDb.add(id, "once", timeMs.toString(), ids, patch);
  console.log(`套 [蜊倡匱逋ｻ骭ｲ] ${date.toLocaleString()} -> ${ids.join(",")}`);

  // 庁 縲占ｿｽ蜉縲第ｬ｡縺ｮ00遘・繝昴・繝ｪ繝ｳ繧ｰ)繧医ｊ繧ょ燕縺ｫ螳溯｡後☆縺ｹ縺咲峩霑代・繧ｸ繝ｧ繝悶↑繧峨√＞縺ｾ逶ｴ謗･繧ｿ繧､繝槭・繧剃ｻ墓寺縺代ｋ
  const nextCronTime = Math.ceil(now / 60000) * 60000; 
  if (timeMs < nextCronTime) {
    setTimeout(() => {
      // 螳牙・陬・ｽｮ・育匱轣ｫ譎ゅ↓DB縺ｫ縺ゅｋ縺狗｢ｺ隱搾ｼ・
      if (scheduleDb.getAll().some((r: any) => r.id === id)) {
        deviceManager.updateDesired(ids, patch);
        scheduleDb.delete(id);
      }
    }, timeMs - now);
  }
};

/**
 * 庁 繧ｹ繧ｱ繧ｸ繝･繝ｼ繝ｩ繝ｼ縺ｮ蛻晄悄蛹悶→繝昴・繝ｪ繝ｳ繧ｰ
 */
export const initCronJobs = () => {
  console.log("竢ｰ 繧ｹ繧ｱ繧ｸ繝･繝ｼ繝ｩ繝ｼ繧貞・譛溷喧");

  const cronJobs = scheduleDb.getAllCron();
  cronJobs.forEach(row => {
    const ids = JSON.parse(row.target_ids);
    const patch = JSON.parse(row.patch);
    
    // 庁 襍ｷ蜍墓凾縺ｮ蠕ｩ蜈・〒繧・ap縺ｫ菫晏ｭ倥☆繧・
    const job = Bun.cron(row.trigger, () => {
      deviceManager.updateDesired(ids, patch);
    });
    activeCronJobs.set(row.id, job);
  });

  // 2. 豈主・縺ｮ繝√ぉ繝・き繝ｼ・亥腰逋ｺ繧ｸ繝ｧ繝也畑・・
  // 庁 繝｡繝｢繝ｪ荳翫・驟榊・縺ｯ荳蛻・ｮ｡逅・○縺壹∵ｯ主屓逶ｴ謗･DB縺ｫ縲御ｻ翫ｄ繧九∋縺堺ｻ穂ｺ九≠繧具ｼ溘阪→閨槭￥
  Bun.cron("* * * * *", () => {
    const now = Date.now();
    const nextMinute = now + 60000;

    // 驕主悉縺ｮ螳溯｡梧ｼ上ｌ・医し繝ｼ繝舌・蛛懈ｭ｢荳ｭ縺ｪ縺ｩ・峨ｂ蜷ｫ繧√※縲√％縺ｮ1蛻・ｻ･蜀・↓螳溯｡後☆縺ｹ縺阪ず繝ｧ繝悶ｒ蜿門ｾ・
    const dueJobs = scheduleDb.getDueOnce(nextMinute);

    dueJobs.forEach(row => {
      const jobTime = parseInt(row.trigger, 10);
      const ids = JSON.parse(row.target_ids);
      const patch = JSON.parse(row.patch);

      // 螳溯｡梧凾蛻ｻ縺ｾ縺ｧ縺ｮ繝・ぅ繝ｬ繧､繧定ｨ育ｮ暦ｼ磯℃蜴ｻ縺ｮ繧ｸ繝ｧ繝悶↑繧・0 縺ｫ縺ｪ繧雁叉譎ょｮ溯｡後＆繧後ｋ・・
      const delay = Math.max(0, jobTime - now);

      setTimeout(() => {
        // 庁 鬲斐・60遘帝俣蟇ｾ遲・ 逋ｺ轣ｫ縺励◆迸ｬ髢薙↓縲後∪縺DB縺ｫ蟄伜惠縺吶ｋ縺九阪ｒ譛邨ら｢ｺ隱・
        const stillExists = scheduleDb.getAll().some((r: any) => r.id === row.id);
        
        if (stillExists) {
          logDb.add("INFO", "CRON", "EXECUTE_ONCE", { trigger: row.trigger, patch }, ids.join(","));
          deviceManager.updateDesired(ids, patch);
          scheduleDb.delete(row.id); // 螳溯｡後′螳御ｺ・＠縺溽椪髢薙↓DB縺九ｉ豸医☆
        } else {
          // 蠕・ｩ滉ｸｭ縺ｫ繝ｦ繝ｼ繧ｶ繝ｼ縺ｫ繧医▲縺ｦ繧ｭ繝｣繝ｳ繧ｻ繝ｫ縺輔ｌ縺溷ｴ蜷医・菴輔ｂ縺励↑縺・
          console.log(`尅 [繧ｭ繝｣繝ｳ繧ｻ繝ｫ讀懃衍] 蠕・ｩ滉ｸｭ縺ｫ蜑企勁縺輔ｌ縺溷腰逋ｺ繧ｸ繝ｧ繝悶・螳溯｡後ｒ蜿悶ｊ繧・ａ縺ｾ縺励◆: ${row.id}`);
        }
      }, delay);
    });
  });

  Bun.cron("0 3 * * *", () => {
    logDb.cleanupOldLogs(30);
    console.log("ｧｹ [繝ｭ繧ｰ繝ｭ繝ｼ繝・・繧ｷ繝ｧ繝ｳ] 30譌･莉･荳雁燕縺ｮ繝ｭ繧ｰ繧貞炎髯､縺励∪縺励◆");
  });
};

  getDueOnce: (targetTimeMs: number) => {

  },

  getAll: () => {
    return db.query("SELECT * FROM schedules").all() as any[];
  }
};

    translate: (v: any) => string;
    parse: (raw: string) => any;
  }>;
  readonly isError?: (raw: string) => boolean;
  readonly ping?: { payload: string; protocol?: ProtocolType; };
  readonly events?: Record<string, {
    translate: (v: any) => string;
  }>;
  readonly toleratedStates?: Record<string, Record<string, string[]>>;
}

/**
 * 邏皮ｲ九↑螳夂ｾｩ繝・・繧ｿ縺九ｉ縲［anager縺梧桶縺医ｋ AVDevice 繧ｪ繝悶ず繧ｧ繧ｯ繝医ｒ蜷域・縺吶ｋ
 */
export const createDevice = <T>(
  def: DeviceDefinition<T>,
  config: { id: string; ip: string; port?: number }
): AVDevice<T> => {
  const port = config.port ?? def.defaultPort;

  return {
    id: config.id,
    type: def.type,
    zodSchema: def.zodSchema,
    // states 縺ｮ繧ｭ繝ｼ縺九ｉ閾ｪ蜍慕噪縺ｫ statusKeys 繧堤函謌・[cite: 51, 68, 83]
    statusKeys: Object.keys(def.states),
    checkError(raw: string) {
      return def.isError ? def.isError(raw) : false;
    },
    getPingTask() {
      if (!def.ping) return null;
      return TaskFactory.create(def.ping.protocol || def.protocol, { ...config, port, payload: def.ping.payload });
    },
    toleratedStates: def.toleratedStates,

    _schema: undefined as T,

    translate(key, value) {
      // 1. 蝠上＞蜷医ｏ縺帙さ繝槭Φ繝峨・蜈ｱ騾壼・逅・
      if (value === "?") {
        const handler = def.states[key];
        if (!handler) return null;
        const payload = handler.translate("?");
        return TaskFactory.create(def.protocol, { ...config, port, payload });
      }

      // 2. 騾壼ｸｸ繧ｳ繝槭Φ繝峨・蜃ｦ逅・(states 縺ｾ縺溘・ events 縺九ｉ讀懃ｴ｢)
      const handler = (def.states as any)[key] || (def.events as any)?.[key];
      const payload = handler?.translate(value);
      
      return payload 
        ? TaskFactory.create(def.protocol, { ...config, port, payload: String(payload) })
        : null;
    },

    parseResponse(key, raw) {
      return def.states[key]?.parse(raw) ?? raw.trim();
    }
  };
};

// src/lib/protocols.ts

export interface ControlTask {
  readonly deviceId: string;
// 庁 霑比ｿ｡繧呈悄蠕・＠縺ｪ縺・ｴ蜷茨ｼ・DP縺ｪ縺ｩ・峨・ null 繧定ｿ斐☆繧医≧縺ｫ螟画峩
  execute(): Promise<string | null>;
}

// Bun繝阪う繝・ぅ繝悶↑TCP騾壻ｿ｡ (PJLink遲・
export class TcpTask implements ControlTask {
  constructor(
    public readonly deviceId: string,
    private ip: string,
    private port: number,
    private payload: string,
    private timeout = 2000
  ) {}

  execute(): Promise<string> { 
    return new Promise((resolve, reject) => {
      let response = "";
      let isTimedOut = false;
      let socketRef: any = null;

      console.log(`[TCP SEND] ${this.deviceId} (${this.ip}) >> ${this.payload.trim()}`);

      // 1. 騾壻ｿ｡蜈ｨ菴難ｼ医ワ繝ｳ繝峨す繧ｧ繧､繧ｯ繧貞性繧・峨・繧ｿ繧､繝繧｢繧ｦ繝医ち繧､繝槭・繧呈怙蛻昴↓莉墓寺縺代ｋ
      const timer = setTimeout(() => {
        isTimedOut = true;
        if (socketRef) socketRef.end(); // 縺吶〒縺ｫ繧ｽ繧ｱ繝・ヨ縺檎函謌舌＆繧後※縺・ｌ縺ｰ遐ｴ譽・
        reject(new Error("TCP Connection/Read Timeout"));
      }, this.timeout);

      Bun.connect({
        hostname: this.ip,
        port: this.port,
        socket: {
          data(s, data) {
            response += new TextDecoder().decode(data);
            s.end(); // 蠢懃ｭ斐ｒ蜿励￠蜿悶▲縺溘ｉ蜊ｳ蛻・妙
          },
          close() { 
            if (!isTimedOut) {
              clearTimeout(timer);
              resolve(response);
            }
          },
          error(s, err) { 
            if (!isTimedOut) {
              clearTimeout(timer);
              reject(err);
            }
          }
        }
      }).then(socket => {
        socketRef = socket;
        
        // 2. 繧ｿ繧､繝繧｢繧ｦ繝亥ｾ後↓郢九′縺｣縺ｦ縺励∪縺｣縺溷ｴ蜷医・繧ｴ繝ｼ繧ｹ繝医さ繝槭Φ繝蛾∽ｿ｡繧帝亟豁｢
        if (isTimedOut) {
          socket.end();
          return;
        }
        
        socket.write(this.payload);
      }).catch(err => {
        if (!isTimedOut) {
          clearTimeout(timer);
          reject(err);
        }
      });
    });
  }
}

// Bun繝阪う繝・ぅ繝悶↑UDP騾∽ｿ｡ (BrightSign遲・

// 庁 1. HTTP繧ｿ繧ｹ繧ｯ縺ｮ霑ｽ蜉
export class HttpTask implements ControlTask {
  constructor(
    public readonly deviceId: string,
    private ip: string,
    private port: number,
    private payload: string,
    private timeout = 2000
  ) {}

  async execute(): Promise<string | null> {
    // payload縺ｮ繝輔か繝ｼ繝槭ャ繝井ｾ・ 
    // "GET /api/status" 
    // "POST /api/power {\"state\":\"on\"}"
    const parts = this.payload.trim().split(" ");
    const method = ["GET", "POST", "PUT", "DELETE"].includes(parts[0] ?? "") ? parts[0] : "GET";
    
    // 繝｡繧ｽ繝・ラ縺檎怐逡･縺輔ｌ縺ｦ "/api/..." 縺九ｉ蟋九∪縺｣縺溷ｴ蜷医・ GET 縺ｨ縺ｿ縺ｪ縺・
    const pathStr = method === "GET" && parts[0] !== "GET" ? parts[0] : parts[1];
    const path = pathStr?.startsWith("/") ? pathStr : `/${pathStr || ""}`;
    
    // 谿九ｊ縺ｮ譁・ｭ怜・繧谷SON遲峨・繝懊ョ繧｣縺ｨ縺励※謇ｱ縺・
    const body = parts.slice(2).join(" ") || undefined;

    const url = `http://${this.ip}:${this.port}${path}`;
    console.log(`[HTTP SEND] ${this.deviceId} >> ${method} ${url} ${body || ""}`);

    // Bun縺ｮfetch縺ｯAbortController縺ｧ繧ｿ繧､繝繧｢繧ｦ繝医ｒ蛻ｶ蠕｡縺ｧ縺阪ｋ
    const controller = new AbortController();
    const timer = setTimeout(() => controller.abort(), this.timeout);

    try {
      const response = await fetch(url, {
        method,
        body,
        signal: controller.signal,
        headers: body ? { "Content-Type": "application/json" } : undefined
      });

      clearTimeout(timer);

      if (!response.ok) {
        // HTTP繧ｹ繝・・繧ｿ繧ｹ繧ｨ繝ｩ繝ｼ(404繧・00)縺ｯ縲‥evice.checkError縺ｧ蠑ｾ縺代ｋ繧医≧縺ｫ迚ｹ螳壹・繝輔か繝ｼ繝槭ャ繝医〒霑斐☆
        return `HTTP_ERROR:${response.status}`;
      }

      // 繝ｬ繧ｹ繝昴Φ繧ｹ繝懊ョ繧｣繧偵ユ繧ｭ繧ｹ繝医→縺励※霑斐☆
      return await response.text();
    } catch (err: any) {
      clearTimeout(timer);
      if (err.name === 'AbortError') {
        throw new Error("HTTP Timeout");
      }
      throw err; // 繝阪ャ繝医Ρ繝ｼ繧ｯ繧ｨ繝ｩ繝ｼ縺ｨ縺励※荳翫・ catch 縺ｫ諡ｾ繧上○繧・
    }
  }
}

export type ProtocolType = "TCP" | "UDP" | "HTTP"; // "HTTP" 繧定ｿｽ蜉

// 庁 3. 繝輔ぃ繧ｯ繝医Μ縺ｮ譖ｴ譁ｰ
export const TaskFactory = {
  create: (
    protocol: ProtocolType,
    config: { id: string; ip: string; port: number; payload: string }
  ): ControlTask => {
    switch (protocol) {
      case "TCP":
        return new TcpTask(config.id, config.ip, config.port, config.payload);
      case "UDP":
        return new UdpTask(config.id, config.ip, config.port, config.payload);
      case "HTTP":
        return new HttpTask(config.id, config.ip, config.port, config.payload); // 霑ｽ蜉
      default:
        throw new Error(`Unsupported protocol: ${protocol}`);
    }
  },
};

  addScheduleMessageSchema,
  getSchedulesMessageSchema,
  deleteScheduleMessageSchema, // 庁 霑ｽ蜉
]);

// --- 繝ｫ繝ｼ繧ｿ繝ｼ譛ｬ菴・---

export const handleClientMessage = async (
  rawMessage: string, 
  broadcastUpdate: () => void,
  sendResponse: (data: any) => void 
) => {
  try {
    const json = JSON.parse(rawMessage);
    const result = wsClientMessageSchema.safeParse(json);

    if (!result.success) {
      console.warn("笞・・荳肴ｭ｣縺ｪ繝｡繝・そ繝ｼ繧ｸ:", result.error.issues);
      return;
    }

    const msg = result.data;

    switch (msg.type) {
      case "UPDATE":
        logDb.add("INFO", "USER", "UPDATE_DESIRED", { patch: msg.patch }, msg.ids.join(","));
        await deviceManager.updateDesired(msg.ids, msg.patch, msg.options);
        broadcastUpdate();
        break;

      case "ADD_SCHEDULE":
        logDb.add("INFO", "USER", "ADD_SCHEDULE", { trigger: msg.trigger, patch: msg.patch, type: msg.scheduleType }, msg.ids.join(","));
        const { scheduleOnce, scheduleRecurring } = await import("./cron");
        if (msg.scheduleType === "once") {
          scheduleOnce(new Date(msg.trigger), msg.ids, msg.patch);
        } else {
          scheduleRecurring(msg.trigger, msg.ids, msg.patch);
        }
        break;

      case "GET_SCHEDULES":
        let rows: any[];
        if (msg.start && msg.end) {
          const startMs = new Date(msg.start).getTime();
          const endMs = new Date(msg.end).getTime();
          rows = scheduleDb.getRange(startMs, endMs);
        } else {
          rows = scheduleDb.getAll();
        }

        const schedules = rows.map((row: any) => ({
          id: row.id,
          type: row.type,
          trigger: row.trigger,
          ids: JSON.parse(row.target_ids),
          patch: JSON.parse(row.patch)
        }));
        
        sendResponse({ type: "SCHEDULE_LIST", data: schedules });
        break;

      case "DELETE_SCHEDULE":
        logDb.add("INFO", "USER", "DELETE_SCHEDULE", { schedule_id: msg.id });
        // 庁 繝・・繧ｿ繝吶・繧ｹ縺九ｉ蜑企勁
        scheduleDb.delete(msg.id);
        console.log(`卵・・[繧ｹ繧ｱ繧ｸ繝･繝ｼ繝ｫ蜑企勁] ID: ${msg.id}`);
        // 蜑企勁螳御ｺ・ｒ騾夂衍
        sendResponse({ type: "SCHEDULE_DELETED", id: msg.id });
        break;
    }
  } catch (error) {
    console.error("笶・Router Error:", error);
  }
};

// scripts/export-types.ts
import { zodToTs, createTypeAlias, printNode } from "zod-to-ts";
import { zodToJsonSchema } from "zod-to-json-schema";
import { z } from "zod";
import { wsClientMessageSchema } from "../lib/ws-router";
import { inventory } from "../lib/inventory";

console.log("噫 繧ｯ繝ｩ繧､繧｢繝ｳ繝亥髄縺代せ繧ｭ繝ｼ繝槭ｒ繧ｨ繧ｯ繧ｹ繝昴・繝井ｸｭ...");

const deviceSchemas = new Map<string, z.ZodObject<any>>();
for (const device of inventory) {
  if (device.zodSchema && !deviceSchemas.has(device.type)) {
    deviceSchemas.set(device.type, device.zodSchema);
  }
}

// ==========================================
// 1. TypeScript迺ｰ蠅・髄縺・(.ts 縺ｮ邨・∩遶九※)
// ==========================================
// v1.2.0縺ｮ莉墓ｧ倥↓蜷医ｏ縺帙∫ｬｬ2蠑墓焚縺ｫ逶ｴ謗･譁・ｭ怜・・亥梛蜷搾ｼ峨ｒ貂｡縺励∪縺・
const { node: msgNode } = zodToTs(wsClientMessageSchema, "ClientMessage");
const msgAlias = createTypeAlias(msgNode, "ClientMessage");

let tsCode = `
/**
 * 笞・・閾ｪ蜍慕函謌舌ヵ繧｡繧､繝ｫ: 繝舌ャ繧ｯ繧ｨ繝ｳ繝峨・Zod繧ｹ繧ｭ繝ｼ繝槭°繧臥函謌舌＆繧後※縺・∪縺吶・
 * 逶ｴ謗･邱ｨ髮・○縺壹√ヰ繝・け繧ｨ繝ｳ繝峨・Zod繧ｹ繧ｭ繝ｼ繝槭ｒ譖ｴ譁ｰ縺励※蜀咲函謌舌＠縺ｦ縺上□縺輔＞縲・
 */

// --- 騾壻ｿ｡繝｡繝・そ繝ｼ繧ｸ縺ｮ蝙・---
export ${printNode(msgAlias)}

// --- 蜷・ョ繝舌う繧ｹ縺ｮ繝励Ο繝代ユ繧｣(迥ｶ諷・蝙・---
`;

const stateTypeNames: string[] = [];

for (const [deviceType, schema] of deviceSchemas.entries()) {
  const typeName = `${deviceType}State`;
  stateTypeNames.push(typeName);

  // 縺薙％繧ょ酔讒倥↓隨ｬ2蠑墓焚繧呈枚蟄怜・縺ｫ縺励∪縺・
  const { node } = zodToTs(schema, typeName);
  const alias = createTypeAlias(node, typeName);
  
  tsCode += `export ${printNode(alias)}\n\n`;
}

if (stateTypeNames.length > 0) {
  tsCode += `// 豎守畑逧・↓菴ｿ縺医ｋ繧医≧縺ｫ蜈ｨ繝・ヰ繧､繧ｹ縺ｮ迥ｶ諷九ｒ縺ｾ縺ｨ繧√◆Union蝙欺n`;
  tsCode += `export type AnyDeviceState = ${stateTypeNames.join(" | ")};\n`;
}

await Bun.write("./frontend-protocol.ts", tsCode.trim());
console.log("笨・TypeScript蝙句ｮ夂ｾｩ繧貞・蜉帙＠縺ｾ縺励◆ (frontend-protocol.ts)");

// ==========================================
// 2. 莉冶ｨ隱樒腸蠅・髄縺・(.json 縺ｮ邨・∩遶九※)
// ==========================================
const definitions: Record<string, any> = {};

const msgJson = zodToJsonSchema(wsClientMessageSchema, { $refStrategy: "none" }) as any;
delete msgJson.$schema;
definitions["ClientMessage"] = msgJson;

for (const [deviceType, schema] of deviceSchemas.entries()) {
  const typeName = `${deviceType}State`;
  const devJson = zodToJsonSchema(schema, { $refStrategy: "none" }) as any;
  delete devJson.$schema;
  definitions[typeName] = devJson;
}

const finalJsonSchema = {
  $schema: "http://json-schema.org/draft-07/schema#",
  description: "Simple AV Manager Protocol Schema",
  $ref: "#/definitions/ClientMessage",
  definitions: definitions
};

await Bun.write("./protocol-schema.json", JSON.stringify(finalJsonSchema, null, 2));
console.log("笨・JSON Schema繧貞・蜉帙＠縺ｾ縺励◆ (protocol-schema.json)");
