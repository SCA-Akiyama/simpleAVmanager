    },
  }

// src/lib/factory.ts
import { TaskFactory, type ProtocolType } from "./protocols";
import type { AVDevice } from "./types";

// 繝・ヰ繧､繧ｹ縺ｮ縲悟ｮ夂ｾｩ縲阪・蝙九ゆｽ懆・・縺薙ｌ縺縺代ｒ譖ｸ縺上・
export interface DeviceDefinition<T = any> {
  readonly type: string;
  readonly protocol: ProtocolType;
  readonly defaultPort: number;
  readonly states: Record<string, {
    translate: (v: any) => string;
    parse: (raw: string) => any;
  }>;
  readonly events?: Record<string, {
    translate: (v: any) => string;
  }>;
  readonly toleratedStates?: Record<string, Record<string, string[]>>;
}

/**
 * 邏皮ｲ九↑螳夂ｾｩ繝・・繧ｿ縺九ｉ縲［anager縺梧桶縺医ｋ AVDevice 繧ｪ繝悶ず繧ｧ繧ｯ繝医ｒ蜷域・縺吶ｋ
 */
export const createDevice = <T>(
  def: DeviceDefinition<T>,
  config: { id: string; ip: string; port?: number }
): AVDevice<T> => {
  const port = config.port ?? def.defaultPort;

  return {
    id: config.id,
    type: def.type,
    // states 縺ｮ繧ｭ繝ｼ縺九ｉ閾ｪ蜍慕噪縺ｫ statusKeys 繧堤函謌・[cite: 51, 68, 83]
    statusKeys: Object.keys(def.states),
    toleratedStates: def.toleratedStates,

    _schema: undefined as T,

    translate(key, value) {
      // 1. 蝠上＞蜷医ｏ縺帙さ繝槭Φ繝峨・蜈ｱ騾壼・逅・
      if (value === "?") {
        const handler = def.states[key];
        if (!handler) return null;
        const payload = handler.translate("?");
        return TaskFactory.create(def.protocol, { ...config, port, payload });
      }

      // 2. 騾壼ｸｸ繧ｳ繝槭Φ繝峨・蜃ｦ逅・(states 縺ｾ縺溘・ events 縺九ｉ讀懃ｴ｢)
      const handler = (def.states as any)[key] || (def.events as any)?.[key];
      const payload = handler?.translate(value);
      
      return payload 
        ? TaskFactory.create(def.protocol, { ...config, port, payload: String(payload) })
        : null;
    },

    parseResponse(key, raw) {
      return def.states[key]?.parse(raw) ?? raw.trim();
    }
  };
};

import { inventory } from "./inventory";

// 繝｡繝｢繝ｪ荳翫・迥ｶ諷狗ｮ｡逅・
const desiredMap = new Map<string, any>();
const actualMap = new Map<string, any>();

// 迥ｶ諷九′貅縺溘＆繧後※縺・ｋ縺句愛螳壹☆繧狗ｴ皮ｲ矩未謨ｰ
const isStateMet = (device: AVDevice, key: string, desired: any, actual: any): boolean => {
  // device.toleratedStates 縺悟ｭ伜惠縺励↑縺・ｩ滓攝縺ｮ蝣ｴ蜷医・縲√％縺薙′ undefined 縺ｫ縺ｪ繧・
  const tolerated = device.toleratedStates?.[key]?.[desired];
  
  if (tolerated && Array.isArray(tolerated)) {
    return tolerated.includes(actual);
  }
  
  // 險ｱ螳ｹ繝・・繧ｿ縺後↑縺・ｩ滓攝縺ｯ縲∽ｻ翫∪縺ｧ騾壹ｊ縺ｮ螳悟・荳閾ｴ縺ｧ蛻､螳夲ｼ医ヵ繧ｩ繝ｼ繝ｫ繝舌ャ繧ｯ・・
  return desired === actual;
};

export const deviceManager = {
  /**
   * 襍ｷ蜍墓凾縺ｫSQLite縺九ｉ菫晏ｭ倥＆繧後◆縲檎炊諠ｳ縲阪・迥ｶ諷九ｒ蠕ｩ蜈・☆繧・
   */
  init: () => {
    const savedStates = stateDb.loadAll();
    savedStates.forEach(row => {
      const current = desiredMap.get(row.device_id) || {};
      desiredMap.set(row.device_id, { 
        ...current, 
        [row.key]: JSON.parse(row.value) 
      });
    });
    console.log(`[DB] ${savedStates.length} 莉ｶ縺ｮ逅・Φ迥ｶ諷九ｒ蠕ｩ蜈・＠縺ｾ縺励◆縲Ａ);
  },

  /**
   * 繝ｦ繝ｼ繧ｶ繝ｼ謫堺ｽ懊ｄAPI縺九ｉ縺ｮ縲檎炊諠ｳ縲阪・譖ｴ譁ｰ繧貞女縺台ｻ倥￠繧・(Fast Path縺ｮ襍ｷ轤ｹ)
   */
  updateDesired: async (ids: string[], patch: any, options = { immediateOnly: false }) => {
    for (const id of ids) {
      const next = { ...(desiredMap.get(id) || {}), ...patch };
      desiredMap.set(id, next);

      // 繧ｹ繝ｩ繧､繝繝ｼ謫堺ｽ懶ｼ・mmediateOnly・峨〒縺ｪ縺代ｌ縺ｰDB縺ｫ豌ｸ邯壼喧
      if (!options.immediateOnly) {
        Object.keys(patch).forEach(key => {
          stateDb.save(id, key, patch[key]);
        });
      }
    }
    // 謫堺ｽ懊＠縺溘ョ繝舌う繧ｹ縺ｫ蟇ｾ縺励※蜊ｳ蠎ｧ縺ｫ蜷梧悄繧貞ｮ溯｡・(Fast Path)
    await deviceManager.syncMany(ids, false, options.immediateOnly);
  },

  /**
   * 隍・焚繝・ヰ繧､繧ｹ縺ｮ蜷梧悄繧剃ｸｦ蛻怜ｮ溯｡・
   */
  syncMany: async (ids: string[], isSlowPath: boolean, immediateOnly = false) => {
    await Promise.all(ids.map(id => deviceManager.syncOne(id, isSlowPath, immediateOnly)));
  },

  /**
   * 1蜿ｰ縺ｮ繝・ヰ繧､繧ｹ縺ｫ蟇ｾ縺吶ｋ蜷梧悄繝ｭ繧ｸ繝・け縺ｮ譬ｸ蠢・
   */
  syncOne: async (id: string, isSlowPath: boolean, immediateOnly = false) => {
    const device = inventory.find(d => d.id === id) as AVDevice | undefined;
    if (!device) return;

    const statusKeys = device.statusKeys || [];

    // 1. 迴ｾ螳・Actual)縺ｮ譖ｴ譁ｰ (Slow Path譎・
    if (isSlowPath) {
      for (const key of statusKeys) {
        const task = device.translate(key, "?");
        if (!task) continue; // BrightSign縺ｪ縺ｩ縺ｮ縲悟撫縺・↓遲斐∴繧峨ｌ縺ｪ縺・ｩ滓攝縲阪・繧ｹ繧ｭ繝・・縺輔ｌ繧・
        try {
          const raw = await task.execute();
          if (raw) {
            const val = device.parseResponse(key, raw);
            actualMap.set(id, { ...(actualMap.get(id) || {}), [key]: val });
          }
        } catch (e) { console.warn(`[Query Failed] ${id}:${key}`); }
      }
    }

    // 2. 蟾ｮ蛻・・蜷梧悄
    const desired = desiredMap.get(id) || {};
    const actual = actualMap.get(id) || {};
    
    for (const key of Object.keys(desired)) {
      const isStatus = statusKeys.includes(key);

      // Slow Path譎ゅ・縲・vent・育憾諷九く繝ｼ莉･螟厄ｼ峨・蜀埼√・縺励↑縺・
      if (isSlowPath && !isStatus) continue;

      // 笘・､画峩・壼ｮ悟・荳閾ｴ・・==・峨°繧峨（sStateMet 髢｢謨ｰ縺ｧ縺ｮ蛻､螳壹↓鄂ｮ縺肴鋤縺・
      // 縲仙､画峩蜑阪訴f (isStatus && desired[key] === actual[key]) continue;
      // 縲仙､画峩蠕後・
      if (isStatus && isStateMet(device, key, desired[key], actual[key])) continue;

      const task = device.translate(key, desired[key]);
      if (!task) continue;

      try {
        await task.execute();
        
        // 縲舌・繧､繝ｳ繝医薦vent縲√せ繝ｩ繧､繝繝ｼ謫堺ｽ懊√∪縺溘・縲悟撫縺・粋繧上○荳榊庄縺ｮ讖滓攝縲阪↑繧・
        // 騾∽ｿ｡謌仙粥縺励◆迸ｬ髢薙↓ actual 繧呈峩譁ｰ縺吶ｋ (讌ｽ隕ｳ逧・峩譁ｰ)
        const canQuery = device.translate(key, "?") !== null;
        if (!isStatus || immediateOnly || !canQuery) {
          actualMap.set(id, { ...(actualMap.get(id) || {}), [key]: desired[key] });
        }
      } catch (e: any) { console.error(`[Sync Failed] ${id}:${key}`); }
    }
  },

  /**
   * 蜈ｨ繝・ヰ繧､繧ｹ繧貞ｯｾ雎｡縺ｨ縺励◆螳壽悄蜷梧悄 (Slow Path)
   */
  syncAll: async () => {
    const allIds = inventory.map(d => d.id);
    await deviceManager.syncMany(allIds, true);
  },

  /**
   * UI陦ｨ遉ｺ逕ｨ縺ｮ繝・・繧ｿ繧ｻ繝・ヨ繧貞叙蠕・
   */
  getStatus: () => inventory.map(d => ({
    id: d.id,
    desired: desiredMap.get(d.id) || {},
    actual: actualMap.get(d.id) || {}
  }))
};

// src/lib/protocols.ts

export interface ControlTask {
  readonly deviceId: string;
  execute(): Promise<string>;
}

// Bun繝阪う繝・ぅ繝悶↑TCP騾∽ｿ｡ (PJLink遲・
export class TcpTask implements ControlTask {
  constructor(
    public readonly deviceId: string,
    private ip: string,
    private port: number,
    private payload: string,
    private timeout = 2000
  ) {}

  async execute(): Promise<string> { // 謌ｻ繧雁､縺ｮ蝙九ｒ螟画峩
    return new Promise(async (resolve, reject) => {
      let response = "";
      console.log(`[TCP SEND] ${this.deviceId} (${this.ip}) >> ${this.payload.trim()}`);

      try {
        const socket = await Bun.connect({
          hostname: this.ip,
          port: this.port,
          socket: {
            data(s, data) {
              response += new TextDecoder().decode(data);
              s.end(); 
            },
            close() { resolve(response); }, // response 繧定ｿ斐＠縺ｦ隗｣豎ｺ
            error(s, err) { reject(err); }
          }
        });
        socket.write(this.payload);
        setTimeout(() => { socket.end(); resolve(response); }, this.timeout);
      } catch (e) { reject(e); }
    });
  }
}

// Bun繝阪う繝・ぅ繝悶↑UDP騾∽ｿ｡ (BrightSign遲・
const udpOut = await Bun.udpSocket({ port: 0 }); // 騾∽ｿ｡蟆ら畑

export class UdpTask implements ControlTask {
  constructor(
    public readonly deviceId: string,
    private ip: string,
    private port: number,
    private payload: string
  ) {}

  async execute(): Promise<string> {
    udpOut.send(this.payload, this.port, this.ip);
    return ""; // UDP縺ｯ霑比ｺ九′縺ｪ縺・・縺ｧ遨ｺ譁・ｭ励ｒ霑斐☆
  }
}

export type ProtocolType = "TCP" | "UDP"; // 蟆・擂逧・↓ "HTTP" 繧・"SERIAL" 繧定ｿｽ蜉蜿ｯ閭ｽ

export const TaskFactory = {
  create: (
    protocol: ProtocolType,
    config: { id: string; ip: string; port: number; payload: string }
  ): ControlTask => {
    switch (protocol) {
      case "TCP":
        return new TcpTask(config.id, config.ip, config.port, config.payload);
      case "UDP":
        return new UdpTask(config.id, config.ip, config.port, config.payload);
      default:
        throw new Error(`Unsupported protocol: ${protocol}`);
    }
  },
};

/**
 * 1. 繝・ヰ繧､繧ｹ蝗ｺ譛峨・螳壽焚・亥錐蜑咲ｩｺ髢難ｼ峨ｒ荳縺､縺ｫ縺ｾ縺ｨ繧√ｋ

 */
const PJ_CONST = {
  power: {
    ON: "on",
    OFF: "off",
    WARMING: "warming",
    COOLING: "cooling",
  },
  shutter: {
    OPEN: "open",
    CLOSE: "close",
  }
} as const;

/**
 * 2. 螳壽焚繧ｪ繝悶ず繧ｧ繧ｯ繝医°繧芽・蜍慕噪縺ｫ蝙九ｒ謚ｽ蜃ｺ縺吶ｋ
 */
export type PjLinkSchema = {
  power?: typeof PJ_CONST.power[keyof typeof PJ_CONST.power];
  shutter?: typeof PJ_CONST.shutter[keyof typeof PJ_CONST.shutter];
  volume?: number;
};

/**
 * 4. 繝・ヰ繧､繧ｹ螳夂ｾｩ譛ｬ菴・
 */
export const pjLinkDef: DeviceDefinition<PjLinkSchema> = {
  type: "PjLinkDevice",
  protocol: "TCP",
  defaultPort: 4352,
  states: {
    // --- 髮ｻ貅仙宛蠕｡ ---
    power: {
      translate: (v) => v === "?" 
        ? "%1POWR ?\r" 
        : (v === PJ_CONST.power.ON ? "%1POWR 1\r" : "%1POWR 0\r"),
      parse: (raw) => {
        if (raw.includes("POWR=0")) return PJ_CONST.power.OFF;
        if (raw.includes("POWR=1")) return PJ_CONST.power.ON;
        if (raw.includes("POWR=2")) return PJ_CONST.power.COOLING;
        if (raw.includes("POWR=3")) return PJ_CONST.power.WARMING;
        return raw.trim();
      }
    },
    // --- 繧ｷ繝｣繝・ち繝ｼ蛻ｶ蠕｡ ---
    shutter: {
      translate: (v) => v === "?" 
        ? "%1SHUT ?\r" 
        : (v === PJ_CONST.shutter.OPEN ? "%1SHUT 0\r" : "%1SHUT 1\r"),
      parse: (raw) => {
        if (raw.includes("SHUT=10")) return PJ_CONST.shutter.OPEN; // 譏蜒乗ｶ亥悉隗｣髯､
        if (raw.includes("SHUT=11")) return PJ_CONST.shutter.CLOSE; // 譏蜒乗ｶ亥悉
        return raw.trim();
      }
    },
    // --- 髻ｳ驥丞宛蠕｡ ---
    volume: {
      translate: (v) => v === "?" ? "%1VOLM ?\r" : `%1VOLM ${v}\r`,
      parse: (raw) => raw.match(/=(.*)/)?.[1]?.trim() || raw.trim()
    }
  },
  // 5. 逅・Φ縺ｨ迴ｾ螳溘・繧ｮ繝｣繝・・繧貞沂繧√ｋ螳夂ｾｩ
  toleratedStates: {
    power: {
      [PJ_CONST.power.ON]: [PJ_CONST.power.ON, PJ_CONST.power.WARMING],
      [PJ_CONST.power.OFF]: [PJ_CONST.power.OFF, PJ_CONST.power.COOLING]
    }
  }
};

import van from "vanjs-core";

// UI縺檎ｮ｡逅・☆繧句髪荳縺ｮ縲悟崋螳壹阪Μ繧ｹ繝・
export const devices = reactive<any[]>([]);
export const status = van.state("繧ｵ繝ｼ繝舌・謗･邯壻ｸｭ...");

const socket = new WebSocket(`ws://${location.host}/ws`);

socket.onmessage = (event) => {
  try {
    const msg = JSON.parse(event.data);
    
    // --- 1. 蛻晏屓謗･邯壽凾 (INIT): 繝ｪ繧ｹ繝医・譫邨・∩繧貞崋螳壹☆繧・---
    if (msg.type === "INIT") {
      devices.length = 0; // 蠢ｵ縺ｮ縺溘ａ繝ｪ繧ｻ繝・ヨ
      msg.data.forEach((d: any) => {
        // 繧ｪ繝悶ず繧ｧ繧ｯ繝亥・菴薙ｒ繝ｪ繧｢繧ｯ繝・ぅ繝悶↓縺励※霑ｽ蜉
        devices.push(reactive(d));
      });
      status.val = "繧ｪ繝ｳ繝ｩ繧､繝ｳ";
    }

    // --- 2. 迥ｶ諷区峩譁ｰ譎・(SYNC): 荳ｭ霄ｫ縺ｮ蛟､・・esired/actual・峨□縺代ｒ譖ｸ縺肴鋤縺医ｋ ---
    if (msg.type === "SYNC") {
      msg.data.forEach((incoming: any) => {
        const target = devices.find(d => d.id === incoming.id);
        if (target) {
          // 驥崎ｦ・ target 閾ｪ菴薙ｒ蟾ｮ縺玲崛縺医ｋ縺ｮ縺ｧ縺ｯ縺ｪ縺上∽ｸｭ霄ｫ縺ｮ繝励Ο繝代ユ繧｣繧呈峩譁ｰ縺吶ｋ
          // 縺薙ｌ縺ｫ繧医ｊ縲．OM隕∫ｴ・・r・峨・菴懊ｊ逶ｴ縺励′逋ｺ逕溘＠縺ｪ縺上↑繧翫∪縺・
          target.desired = incoming.desired;
          target.actual = incoming.actual;
        }
      });
    }
  } catch (e) {
    console.error("繝・・繧ｿ蜷梧悄繧ｨ繝ｩ繝ｼ:", e);
  }
};

// 謫堺ｽ憺∽ｿ｡髢｢謨ｰ縺ｪ縺ｩ縺ｯ螟画峩縺ｪ縺・
export const sendUpdate = (ids: string[], patch: any, immediate = false) => {
  socket.send(JSON.stringify({ type: "UPDATE", ids, patch, options: { immediateOnly: immediate } }));
};

// src/ui/pages/dashboard.ts
import { list } from "vanjs-ext";
import { tags } from "../tags";
import { devices, sendUpdate } from "../store";

export const Dashboard = () => {
  return tags.table({ class: "striped" },
    tags.thead(
      tags.tr(tags.th("繝・ヰ繧､繧ｹID"), tags.th("逅・Φ"), tags.th("迴ｾ螳・), tags.th("謫堺ｽ・))
    ),
    // list髢｢謨ｰ縺ｯ縲碁・蛻励・隕∫ｴ・・evice繧ｪ繝悶ず繧ｧ繧ｯ繝茨ｼ峨阪→DOM・・r・峨ｒ1蟇ｾ1縺ｧ邨舌・縺､縺代∪縺・
    list(tags.tbody(), devices, (device) => 
      tags.tr(
        tags.td(device.val.id),
        // () => 縺ｧ蝗ｲ繧縺薙→縺ｧ縲∽ｸｭ霄ｫ縺ｮ繝励Ο繝代ユ繧｣縺梧峩譁ｰ縺輔ｌ縺滓凾縺縺代％縺薙′譖ｸ縺肴鋤繧上ｊ縺ｾ縺・
        tags.td(() => tags.span({ class: "badge" }, device.val.desired.power || "-")),
        tags.td(() => tags.span({ class: "badge" }, device.val.actual.power || "-")),
        tags.td(
          tags.button({
            onclick: () => sendUpdate([device.val.id], { power: "on" }),
            class: "outline"
          }, "ON"),
          tags.button({
            onclick: () => sendUpdate([device.val.id], { power: "off" }),
            class: "outline secondary"
          }, "OFF")
        )
      )
    )
  );
};
